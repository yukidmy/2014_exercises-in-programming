  76 void getstate(vector2* ret, int* isfire){
  77   static vector2 pos = {SCREENWIDTH/2-1, SCREENHEIGHT-3}; 
  78   *isfire = 0;
  79   int in = gameioread();
  80   switch(in){
  81   case 'f':
  82   case 'F':
  83     pos.x++;
  84     break;
  85   case 'b':
  86   case 'B':
  87     pos.x--;
  88     break;
  89   case 'p':
  90   case 'P':
  91     pos.y--;
  92     break;
  93   case 'n':
  94   case 'N':
  95     pos.y++;
  96     break;
  97   case ' ':
  98     *isfire = 1;
  99   } 
 100   *ret = pos;
 101 }

 113 void gameloop(void){
 114   int speedlevel = 8; 
 115   int time=0;
 116   vector2 tmpv;
 117   event* e = getnextevent();
 118   while(1){
 119     gameiousleep(30000);
 120     if(time%speedlevel == 0){
 121       scrollwall();
 122       while(e!= NULL && e->time*speedlevel < time){
 123         addmo(e->mobj, e->pos, e->vel);
 124         e = getnextevent();
 125       }
 126       score++;
 127     }
 128     getstate(&mypos, &isfire);
 129     gameioput(SCREENWIDTH+3,1,'0'+(int)mypos.x/10);
 130     gameioput(SCREENWIDTH+4,1,'0'+(int)mypos.x%10);
 131     gameioput(SCREENWIDTH+6,1,'0'+(int)mypos.y/10);
 132     gameioput(SCREENWIDTH+7,1,'0'+(int)mypos.y%10);
 133     putwalls(SCREENHEIGHT);
 134     putmo();
 135     onestepmo();
 136     if(checkcollision(mypos, objp[1])!=NONOBJ){
 137       break;
 138     }
 139     if(isfire){
 140       tmpv = mypos;
 141       tmpv.x +=1;
 142       tmpv.y -=2;
 143       addmo(mobjp[1], tmpv, makevector2(0,-1));
 144     }
 145     mobjp[2]->pos = mypos;
 146     dispm(mobjp[2]);
 147     show();
 148     time++;
 149   }
 150 }
